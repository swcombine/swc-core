(defun start
  (say "Alright me, old chiina? You got any bees and honey for you're Denny Glaver?")
  (add-response "Excuse me?" excuse)
  (add-response "You 'avin a bath?" laugh)
  (add-response "Yeah, ok. I see the white men coming." leave)
)

(defun leave
  (say "You scapa now, y'hear! Its time to go, yeah?")
)

(defun excuse
  (say "You, geezer, you're a right ancient greek. Gotta Botany Bay, the pork chops are after me. Lor' luv a duck! Meet up later, yeah. Know what I mean?")
  (add-response "No actually, I have no clue what you're on about." clue)
  (cond
    [(eq? (timeofday) "Evening")
     (add-response "They won't catch you in this darkness though." dark)])
)

(defun laugh
  (say "Awright geeezzaa! No mate, just need some Cyril Ash fer da frod an' toad. Sorted mate.")
  (add-response "No actually, I have no clue what you're on about." clue)
  (add-response "Lawd above! Sure, 'ave some beans n 'oney then, innit." money)
  (add-response "I'm getting a headache. See ya." leave)
)

(defun cops
  (say "Blimey! well duh. Got me some dogs dinner clothes from what lady. Nuff said, yeah?")
  (add-response "Thats it, my mind has been blown." clue)
)

(defun money
  (say "Lor' luv a duck! cheers mate, michael owen ter spend i' on some ladies. Know what I mean?")
  (add-response "Yeah, enjoy yourself." leave)
  (add-response "Not really." clue)
)

(defun clue
  (say "Sheesh, I'll speak slowly for the stupid then. I'm just on my way outta here, with these nice new things I got from that lady over there. Do you understand? Or would you rather I speak in Huttese?")
  (add-response "Ah, much better. I can understand you now." finally)
  (add-response "You're still speaking in another language. I'm off." leave)
)

(defun finally
  (say "Wow, well done for the hard of hearing. Anyway, gotta run, places to be, places not to be. Catch my drift?")
)

(defun dark
  (say "Lor' luv a duck! yeah, all da be'er ter do me drives me bizurke 'uh. Know what I mean?")
)
